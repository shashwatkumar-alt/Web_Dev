|->  +  -> addition
|->  -  -> subtraction
|->  *  -> multiplication
|->  /  -> division
|->  || -> logical OR                    requires two conditions and the true one is the output.
|->  && -> logical AND

|->  0
|->  false
|->  undefined
|->  NaN
|->  null
|->  ""
|->  document.all
